# 单例线程池

```java
package com.iflytek.devops.common.utils;


import java.util.concurrent.*;

public class ThreadPool {

    private final ExecutorService executorService;

    private ThreadPool() {
        int core = Runtime.getRuntime().availableProcessors();
        int maxCore = core * 2;
        int keepAliveTime = 60;
        int capacity = 50000;
        LinkedBlockingQueue<Runnable> blockingQueue = new LinkedBlockingQueue<>(capacity);
        ThreadFactory factory = Executors.defaultThreadFactory();
        executorService = new ThreadPoolExecutor(core, maxCore, keepAliveTime, TimeUnit.MINUTES, blockingQueue, factory);
    }

    public static ThreadPool getInstance() {
        return ThreadPoolHolder.INSTANCE;
    }

    private static class ThreadPoolHolder {
        private static final ThreadPool INSTANCE = new ThreadPool();
    }

    public void execute(Runnable task) {
        executorService.execute(task);
    }

    public <T> Future<T> submit(Runnable task, T clz) {
        return executorService.submit(task, clz);
    }

    public <T> Future<T> submit(Callable<T> call) {
        return executorService.submit(call);
    }

    public Future<?> submit(Runnable task) {
        return executorService.submit(task);
    }

}
```
